-- INNER JOIN
SELECT 
	T1.Nome,
	T1.Idade,
	T2.Logradouro
FROM 
	tbNome AS T1
	INNER JOIN tbEndereco AS T2 ON T1.IdEndereco = T2.Id


-- LEFT JOIN
SELECT 
	T1.Nome,
	T1.Idade,
	T2.Logradouro
FROM 
	tbNome AS T1
	LEFT JOIN tbEndereco AS T2 ON T1.IdEndereco = T2.Id

-- RIGHT JOIN
SELECT 
	T1.Nome,
	T1.Idade,
	T2.Logradouro
FROM 
	tbNome AS T1
	RIGHT JOIN tbEndereco AS T2 ON T1.IdEndereco = T2.Id
